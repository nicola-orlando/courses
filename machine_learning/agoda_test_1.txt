from numpy import sqrt 

def solution(nums, prices):
    
    # Get population 
    N = 0 
    for element in nums: 
        N+= element 
        
    # Get total cost 
    sum = 0
    for index in range(0,len(nums)):
        sum += nums[index]*prices[index]
    average = sum / N 
    
    # Squared residuals 
    res = 0 
    for index in range(0,len(prices)): 
        res += (prices[index] - average)*(prices[index] - average)*nums[index]
    
    print('N = '+str(N))
    print('res = '+str(res))
    print('average = '+str(average))
    
    # Standard deviation 
    std_dev = sqrt(res/N)
    return std_dev

    c_st.sort()
    
    movies_I_can_watch = 1

    print(st_p_d)
    
    # Look at all start times

        I_can_watch = False # I can always watch the first movie

from string import digits

def solution_1(s): 
    # Remove digits 
    remove_digits = str.maketrans('', '', digits)
    s_no_digits = s.translate(remove_digits)

    # Convert string into a list and manipulate swapping roles of upper/lower case  
    list_of_char = [*s_no_digits]
    list_upper_case = []
    list_lower_case = []
    
    # Define two lists, one with upper cases (necessary for reversing) and another with lower cases and empty elements
    for char in list_of_char:
        if char.isupper():
            list_lower_case.append(char.lower())
        if char.islower():
            list_lower_case.append('') 
            list_upper_case.append(char.upper())  

    # Reverse the list 
    list_upper_case = list_upper_case[::-1]
    # Merge the two lists 
    counter_replacements = 0
    for index in range(0,len(list_lower_case)): 
        if list_lower_case[index] == '':
            list_lower_case[index]=list_upper_case[counter_replacements]
            counter_replacements = counter_replacements+1

    # Convert back into a string 
    output = ''.join([str(element) for element in list_lower_case])
    return output
